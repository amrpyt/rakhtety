import { supabase } from '@/lib/supabase/client'
import type { WorkflowStep, WorkflowStepWithEmployee, StepStatus } from '@/types/database.types'
import { ErrorCodes } from '@/types/error-codes.enum'
import { AppError } from '@/lib/errors/app-error.class'

export interface IWorkflowStepRepository {
  findById(id: string): Promise<WorkflowStep | null>
  findByWorkflowId(workflowId: string): Promise<WorkflowStepWithEmployee[]>
  create(data: CreateWorkflowStepData): Promise<WorkflowStep>
  update(id: string, data: UpdateWorkflowStepData): Promise<WorkflowStep>
  updateStatus(id: string, status: StepStatus, completedAt?: string): Promise<WorkflowStep>
  delete(id: string): Promise<void>
  validateTransition(currentStatus: StepStatus, newStatus: StepStatus): boolean
}

export interface CreateWorkflowStepData {
  workflow_id: string
  step_order: number
  name: string
  assigned_to?: string
  fees?: number
  profit?: number
}

export interface UpdateWorkflowStepData {
  name?: string
  assigned_to?: string
  fees?: number
  profit?: number
}

// Valid transitions map
const VALID_TRANSITIONS: Record<StepStatus, StepStatus[]> = {
  pending: ['in_progress'],
  in_progress: ['completed', 'blocked'],
  blocked: ['in_progress'],
  completed: [], // Final state - no transitions allowed
}

export class WorkflowStepRepository implements IWorkflowStepRepository {
  private readonly table = 'workflow_steps'

  // App-level validation of state transitions
  validateTransition(currentStatus: StepStatus, newStatus: StepStatus): boolean {
    const allowedTransitions = VALID_TRANSITIONS[currentStatus]
    return allowedTransitions.includes(newStatus)
  }

  async findById(id: string): Promise<WorkflowStep | null> {
    const { data, error } = await supabase
      .from(this.table)
      .select('*')
      .eq('id', id)
      .single()

    if (error && error.code !== 'PGRST116') {
      throw error
    }

    return data || null
  }

  async findByWorkflowId(workflowId: string): Promise<WorkflowStepWithEmployee[]> {
    const { data, error } = await supabase
      .from(this.table)
      .select('*, assigned_employee:profiles(full_name)')
      .eq('workflow_id', workflowId)
      .order('step_order')

    if (error) throw error
    return data || []
  }

  async create(data: CreateWorkflowStepData): Promise<WorkflowStep> {
    const { data: result, error } = await supabase
      .from(this.table)
      .insert(data)
      .select()
      .single()

    if (error) throw error
    return result
  }

  async update(id: string, data: UpdateWorkflowStepData): Promise<WorkflowStep> {
    const { data: result, error } = await supabase
      .from(this.table)
      .update({ ...data, updated_at: new Date().toISOString() })
      .eq('id', id)
      .select()
      .single()

    if (error) throw error
    return result
  }

  async updateStatus(id: string, status: StepStatus, completedAt?: string): Promise<WorkflowStep> {
    // First fetch current step to validate transition
    const currentStep = await this.findById(id)
    if (!currentStep) {
      throw new AppError({
        code: ErrorCodes.WORKFLOW_STEP_NOT_FOUND,
        message: 'الخطوة غير موجودة',
        statusCode: 404,
      })
    }

    // App-level validation before calling database
    if (!this.validateTransition(currentStep.status, status)) {
      const arabicMessages: Record<string, string> = {
        'pending→completed': 'يجب إكمال الخطوة تدريجياً: انتقل أولاً إلى جاري التنفيذ',
        'pending→blocked': 'لا يمكن حظر خطوة لم تبدأ بعد',
        'completed→pending': 'لا يمكن العودة للخطوة السابقة: اكتملت الخطوة',
        'completed→in_progress': 'لا يمكن تعديل خطوة مكتتملة',
        'completed→blocked': 'لا يمكن تعديل خطوة مكتتملة',
        'blocked→completed': 'يجب فتح الخطوة أولاً قبل الإكمال',
        'blocked→pending': 'تسلسل غير صالح: انتقل أولاً إلى جاري التنفيذ',
      }
      const transitionKey = `${currentStep.status}→${status}`
      const message = arabicMessages[transitionKey] || `لا يمكن الانتقال من ${currentStep.status} إلى ${status}`

      throw new AppError({
        code: ErrorCodes.WORKFLOW_STEP_INVALID_TRANSITION,
        message,
        statusCode: 400,
        context: {
          currentStatus: currentStep.status,
          attemptedStatus: status,
          stepId: id,
        },
      })
    }

    // Build update data
    const updateData: Partial<WorkflowStep> = { status }
    if (completedAt) {
      updateData.completed_at = completedAt
    }

    // Attempt update - PostgreSQL trigger will also validate
    const { data, error } = await supabase
      .from(this.table)
      .update(updateData)
      .eq('id', id)
      .select()
      .single()

    // Handle PostgreSQL trigger exception
    if (error) {
      // Check if it's a trigger violation (PostgreSQL raises exception with specific code)
      if (error.message && error.message.includes('Invalid transition')) {
        // Map PostgreSQL exception to AppError
        const stepName = currentStep.name
        throw new AppError({
          code: ErrorCodes.WORKFLOW_STEP_INVALID_TRANSITION,
          message: `لا يمكن الانتقال من ${currentStep.status} إلى ${status}: اكتملت الخطوة "${stepName}"`,
          statusCode: 400,
          context: { stepId: id, currentStatus: currentStep.status, attemptedStatus: status },
        })
      }
      throw error
    }

    return data
  }

  async delete(id: string): Promise<void> {
    const { error } = await supabase.from(this.table).delete().eq('id', id)
    if (error) throw error
  }
}

export const workflowStepRepository = new WorkflowStepRepository()

import { describe, it, expect } from 'vitest'
import { WorkflowStepRepository, IWorkflowStepRepository } from './workflow-step.repository'
import type { StepStatus } from '@/types/database.types'

describe('WorkflowStepRepository.validateTransition', () => {
  let repository: IWorkflowStepRepository

  beforeEach(() => {
    // Create instance for method testing
    const repo = new WorkflowStepRepository()
    repository = repo as unknown as IWorkflowStepRepository
  })

  describe('Valid transitions', () => {
    it('PENDING → IN_PROGRESS is valid', () => {
      expect(repository.validateTransition('pending', 'in_progress')).toBe(true)
    })

    it('IN_PROGRESS → COMPLETED is valid', () => {
      expect(repository.validateTransition('in_progress', 'completed')).toBe(true)
    })

    it('IN_PROGRESS → BLOCKED is valid', () => {
      expect(repository.validateTransition('in_progress', 'blocked')).toBe(true)
    })

    it('BLOCKED → IN_PROGRESS is valid (unblock)', () => {
      expect(repository.validateTransition('blocked', 'in_progress')).toBe(true)
    })
  })

  describe('Invalid transitions', () => {
    it('PENDING → COMPLETED is invalid (must go through IN_PROGRESS)', () => {
      expect(repository.validateTransition('pending', 'completed')).toBe(false)
    })

    it('PENDING → BLOCKED is invalid', () => {
      expect(repository.validateTransition('pending', 'blocked')).toBe(false)
    })

    it('COMPLETED → PENDING is invalid (final state)', () => {
      expect(repository.validateTransition('completed', 'pending')).toBe(false)
    })

    it('COMPLETED → IN_PROGRESS is invalid (final state)', () => {
      expect(repository.validateTransition('completed', 'in_progress')).toBe(false)
    })

    it('COMPLETED → BLOCKED is invalid (final state)', () => {
      expect(repository.validateTransition('completed', 'blocked')).toBe(false)
    })

    it('BLOCKED → COMPLETED is invalid (must go through IN_PROGRESS)', () => {
      expect(repository.validateTransition('blocked', 'completed')).toBe(false)
    })

    })
  })
