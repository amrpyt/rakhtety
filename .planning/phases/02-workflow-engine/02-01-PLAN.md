- COMPLETED → PENDING (never)
- COMPLETED → IN_PROGRESS (never)
- PENDING → COMPLETED (must go through IN_PROGRESS)
- Any transition from COMPLETED (final state)

```sql
-- Enable pg_trgm extension for potential future text search
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

-- Create enum type for step status if not exists
DO $$ BEGIN
  CREATE TYPE step_status AS ENUM ('pending', 'in_progress', 'completed', 'blocked');
EXCEPTION
  WHEN duplicate_object THEN null;
END $$;

-- Add status column to workflow_steps if not exists
ALTER TABLE public.workflow_steps
  ADD COLUMN IF NOT EXISTS status step_status DEFAULT 'pending';

-- Create trigger function to enforce valid state transitions
CREATE OR REPLACE FUNCTION enforce_workflow_step_transition()
RETURNS TRIGGER AS $$
BEGIN
  -- Allow NULL transitions (no change)
  IF NEW.status IS NULL THEN
    RETURN NEW;
  END IF;

  -- Initial insert: allow any valid initial state
  IF OLD.status IS NULL THEN
    IF NEW.status NOT IN ('pending', 'in_progress', 'blocked') THEN
      RAISE EXCEPTION 'Invalid initial step status: %. Must be pending, in_progress, or blocked', NEW.status;
    END IF;
    RETURN NEW;
  END IF;

  -- PENDING → allowed: IN_PROGRESS
  IF OLD.status = 'pending' AND NEW.status NOT IN ('in_progress') THEN
    RAISE EXCEPTION 'Invalid transition from PENDING to %. Can only transition to IN_PROGRESS', NEW.status;
  END IF;

  -- IN_PROGRESS → allowed: COMPLETED, BLOCKED
  IF OLD.status = 'in_progress' AND NEW.status NOT IN ('completed', 'blocked') THEN
    RAISE EXCEPTION 'Invalid transition from IN_PROGRESS to %. Can only transition to COMPLETED or BLOCKED', NEW.status;
  END IF;

  -- BLOCKED → allowed: IN_PROGRESS (unblock)
  IF OLD.status = 'blocked' AND NEW.status NOT IN ('in_progress') THEN
    RAISE EXCEPTION 'Invalid transition from BLOCKED to %. Can only transition to IN_PROGRESS', NEW.status;
  END IF;

  -- COMPLETED → no transitions allowed (final state)
  IF OLD.status = 'completed' THEN
    RAISE EXCEPTION 'Invalid transition from COMPLETED. Completed steps are final and cannot be changed';
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Create trigger on workflow_steps
DROP TRIGGER IF EXISTS trg_enforce_workflow_step_transition ON public.workflow_steps;
CREATE TRIGGER trg_enforce_workflow_step_transition
  BEFORE UPDATE OF status ON public.workflow_steps
  FOR EACH ROW
  EXECUTE FUNCTION enforce_workflow_step_transition();

-- Add comment for documentation
COMMENT ON FUNCTION enforce_workflow_step_transition() IS
  'Enforces valid workflow step state transitions. Prevents invalid transitions such as completed->pending or pending->completed bypassing in_progress.';
```

This migration:
1. Adds step_status enum type (idempotent - handles case where it already exists)
2. Adds status column with default 'pending' (idempotent)
3. Creates trigger function that enforces:
   - PENDING can only go to IN_PROGRESS
   - IN_PROGRESS can go to COMPLETED or BLOCKED
   - BLOCKED can go back to IN_PROGRESS
   - COMPLETED is final - no transitions allowed
4. Attaches trigger to workflow_steps table

  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  step_name TEXT NOT NULL,
  workflow_type TEXT NOT NULL CHECK (workflow_type IN ('DEVICE_LICENSE', 'EXCAVATION_PERMIT')),
  government_fee DECIMAL(10,2) NOT NULL DEFAULT 0,

  is_active BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE(step_name, workflow_type)
);

-- Create index for faster lookups
CREATE INDEX IF NOT EXISTS idx_workflow_step_configs_lookup
